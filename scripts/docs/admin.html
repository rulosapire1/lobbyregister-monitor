# SSH in GitHub Codespace oder lokal:
cd lobbyregister-monitor

# Nur Daten abrufen:
python scripts/fetch_and_build.py

# Nur Gemini (KI-Filter + Zusammenfassungen):
python scripts/gemini_enrich.py

# Nur Health-Check:
python scripts/health_check.py

# Nur Mail versenden:
python scripts/send_email.py
